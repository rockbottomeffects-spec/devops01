  <script>
    // ====== JavaScript ====== 
    let count = 0; // (45)

    // Get elements (46)
    const counter = document.getElementById("counter"); // (47)
    const increaseBtn = document.querySelector(".increase"); // (48)
    const decreaseBtn = document.querySelector(".decrease"); // (49)
    const resetBtn = document.querySelector(".reset"); // (50)

    // Update counter function (51)
    function updateCounter() { // (52)
      counter.textContent = count; // (53)
      if (count > 0) { // (54)
        counter.style.color = "#4CAF50"; // (55)
      } else if (count < 0) { // (56)
        counter.style.color = "#f44336"; // (57)
      } else { // (58)
        counter.style.color = "#008CBA"; // (59)
      }
    }

    // Increase button click (60)
    increaseBtn.addEventListener("click", () => { // (61)
      count++; // (62)
      updateCounter(); // (63)
    });

    // Decrease button click (64)
    decreaseBtn.addEventListener("click", () => { // (65)
      count--; // (66)
      updateCounter(); // (67)
    });

    // Reset button click (68)
    resetBtn.addEventListener("click", () => { // (69)
      count = 0; // (70)
      updateCounter(); // (71)
    });

    // Initialize display (72)
    updateCounter(); // (73)

    // Extra: Keyboard shortcuts (74)
    document.addEventListener("keydown", (e) => { // (75)
      if (e.key === "ArrowUp") { // (76)
        count++; // (77)
      } else if (e.key === "ArrowDown") { // (78)
        count--; // (79)
      } else if (e.key === "0") { // (80)
        count = 0; // (81)
      }
      updateCounter(); // (82)
    });

    // Extra: Double-click reset (83)
    resetBtn.addEventListener("dblclick", () => { // (84)
      count = 0; // (85)
      alert("Counter reset to zero!"); // (86)
      updateCounter(); // (87)
    });

    // Extra styling on load (88)
    window.addEventListener("load", () => { // (89)
      document.body.style.background = "linear-gradient(to right, #83a4d4, #b6fbff)"; // (90)
    });

    // Log actions (91)
    function logAction(action) { // (92)
      console.log(`Action: ${action}, Current Count: ${count}`); // (93)
    }

    // Add logs to each button (94)
    increaseBtn.addEventListener("click", () => logAction("Increase")); // (95)
    decreaseBtn.addEventListener("click", () => logAction("Decrease")); // (96)
    resetBtn.addEventListener("click", () => logAction("Reset")); // (97)

    // Example timeout (98)
    setTimeout(() => console.log("Counter App Ready!"), 1000); // (99)
  </script> <!-- (100) -->

  <script>
    function addTask(){
      let input=document.getElementById("taskInput");
      let text=input.value.trim();
      if(text==="") return;
      let li=document.createElement("li");
      li.textContent=text;
      li.onclick=()=>li.classList.toggle("done");
      li.ondblclick=()=>li.remove();
      document.getElementById("taskList").appendChild(li);
      input.value="";
    }
  </script>
